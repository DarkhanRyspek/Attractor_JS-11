SELECT *
  FROM OWS.OFFICER O, OWS.OFFICER_GROUP G
 WHERE O.AMND_STATE='A'
       AND UPPER(O.NAME) LIKE '%фсмся%'
       AND O.OFFICER_GROUP__OID=G.ID
       AND G.AMND_STATE='A'
       

SELECT *
  FROM OWS.MENU_ITEM I
 WHERE I.AMND_STATE='A'
       AND UPPER(I.NAME) LIKE '%SMS%MOT%';

SELECT *
  FROM OWS.MENU_TREE T
 WHERE T.AMND_STATE='A'
       AND UPPER(T.NAME) LIKE '%SMS%MOT%';
