SELECT D.SUBITEM__OID, 
       WM_CONCAT(D.DATA)
  FROM OWS.SUBITEM I, OWS.SYS_SUBITEM_DATA D
 WHERE I.MENU_ITEM__OID IN (SELECT T.MENU_ITEM__ID FROM OWS.MENU_TREE T 
                            WHERE T.AMND_STATE='A' AND T.AMND_OFFICER=46726 AND UPPER(T.NAME) LIKE '%SB%LOAN%ROW%')
       AND I.IS_MODAL='N' AND I.SUBITEM_ORDER=3
       AND I.ID=D.SUBITEM__OID
 GROUP BY D.SUBITEM__OID
