SELECT D.TARGET_NUMBER,D.SOURCE_NUMBER,
       D.TRANS_DATE,D.POSTING_DATE,D.POSTING_STATUS,
       D.TRANS_AMOUNT,D.TRANS_CURR,
       OWS.FX.MIDDLE_RATE(NULL,'724',NULL,D.TRANS_CURR,'398',D.TRANS_AMOUNT,D.TRANS_DATE) AS MX_RATE
  FROM OWS.DOC D, OWS.M_TRANSACTION MT
 WHERE D.AMND_STATE='A'
       AND D.IS_AUTHORIZATION='N'
       AND D.POSTING_STATUS = 'P'
       AND D.RETURN_CODE=0
       AND D.ID=MT.DOC__OID
       AND D.SERVICE_CLASS=MT.SERVICE_CLASS
       AND D.TRANS_DETAILS='Списание в связи с закрытие платежной карты'
       --AND D.SOURCE_NUMBER IN (SELECT W.ADD_DATA1 FROM SBERBANK_CUST.TEMP_WT W WHERE W.INFORM = 'NOTPROD' GROUP BY W.ADD_DATA1)
       AND D.ID >= (SELECT MAX(D1.ID) FROM OWS.DOC D1 WHERE D1.AMND_DATE=TO_DATE('20012020','DDMMYYYY'))
       AND D.AMND_DATE >= TO_DATE('27022020 08:01','DDMMYYYY HH24:MI')
 ORDER BY D.TRANS_CURR DESC
