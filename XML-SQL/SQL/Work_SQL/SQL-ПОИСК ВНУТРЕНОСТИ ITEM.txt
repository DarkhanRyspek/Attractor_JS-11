SELECT MI.ID,MI.NAME,
       OWS.XWMENU_ITEM('AMND_OFFICER',MI.AMND_OFFICER) AS AMND_OFFICER_ITEM,
       SI.ID,
       OWS.XWSUBITEM('AMND_OFFICER',SI.AMND_OFFICER) AS AMND_OFFICER_SUBITEM,
       SI.SUBITEM_ORDER,SI.SUBITEM_TYPE,SI.NAME,
       SSI.*
  FROM OWS.MENU_ITEM MI,
       OWS.SUBITEM SI,
       OWS.SYS_SUBITEM_DATA SSI
WHERE MI.AMND_STATE='A'
       --AND MI.NAME='SB.Отчет по Inactiv Account'
       AND SI.AMND_STATE='A'
       AND MI.ID=SI.MENU_ITEM__OID
       AND SI.ID=SSI.SUBITEM__OID
           AND UPPER(SSI.DATA) LIKE '%ADD_INFO_03%'
           AND UPPER(SSI.DATA) LIKE '%ДА%'
           AND UPPER(SSI.DATA) LIKE '%НЕТ%'
ORDER BY MI.ID,SI.SUBITEM_ORDER;

SELECT *
  FROM OWS.MENU_TREE T
START WITH T.AMND_STATE='A' 
            AND T.MENU_ITEM__ID=64262
CONNECT BY PRIOR T.MENU_TREE__OID=T.ID  --UP
